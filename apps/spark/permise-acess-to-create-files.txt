# To permit acess to create or delete files in jupyter notebook from spark, you need to know how is the "UID" and "GID" from container
# Spark master, what the jupyter notebook is running, and run this comand in your machine or VM to allow acess to create and deleted file
# for Container Spark-master

# Run this comand on terminal to acess the bash from container
docker exec -it name-of-container bash 

# To discover your uid and gid on linux terminal
id

# Run this comand on your machine or VM 
sudo chown -R 1000:100 /caminho/para/seu/notebooks # 1000(UID):100(GID) and "/caminho/para/seu/notebooks" its your notebook path
